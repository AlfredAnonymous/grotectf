ACTOR Fist : Weapon
{
  Weapon.Kickback 100
  Obituary "$OB_MPFIST"
  Tag "$TAG_FIST"
  States
  {
  Ready:
    PUNG A 1 A_WeaponReady
    Loop
  Deselect:
    PUNG A 0 A_Lower
	PUNG A 0 A_Lower
	PUNG A 0 A_Lower
	PUNG A 1 A_Lower
    Loop
  Select:
    PUNG A 0 A_Raise
	PUNG A 0 A_Raise
	PUNG A 0 A_Raise
	PUNG A 1 A_Raise
    Loop
  Fire:
    PUNG B 2
    PUNG C 2 A_Saw ("none","player/male/fist",13)
    PUNG D 3
    PUNG C 2
    PUNG B 3 A_ReFire
    Goto Ready
  }
}
